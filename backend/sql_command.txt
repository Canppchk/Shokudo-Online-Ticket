CREATE DATABASE sot;

CREATE TABLE Food (
    Id INT AUTO_INCREMENT PRIMARY KEY,
    Name VARCHAR(255),
    Meal VARCHAR(255),
    Detail TEXT,
    Stock INT,
    Price DECIMAL(10, 2),
    Picture TEXT
);

# Create a user table.

    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    password VARCHAR(255) NOT NULL,
    role INT NOT NULL
